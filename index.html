        function createPopupContent(walk) {

            const date = walk.date;

            

            let buttonsHtml = `<div class="btn-group">`;

            

            if (walk.vk_id && walk.vk_id !== "") {

                buttonsHtml += `<a href="https://vkvideo.ru/video${walk.vk_id}" target="_blank" class="btn btn-vk">▶ VK Видео</a>`;

            }

            if (walk.youtube_id && walk.youtube_id !== "") {

                buttonsHtml += `<a href="https://www.youtube.com/watch?v=${walk.youtube_id}" target="_blank" class="btn btn-youtube">▶ YouTube</a>`;

            }

            if (walk.rutube_id && walk.rutube_id !== "") {

                buttonsHtml += `<a href="https://rutube.ru/video/${walk.rutube_id}/" target="_blank" class="btn btn-rutube">▶ RuTube</a>`;

            }

            buttonsHtml += `</div>`;


            let imageHtml = '';

            if (walk.youtube_id && walk.youtube_id !== "") {

                imageHtml = `<img src="https://img.youtube.com/vi/${walk.youtube_id}/hqdefault.jpg" 

                                  class="video-thumb" 

                                  alt="${walk.title}"

                                  onerror="this.style.display='none'; document.getElementById('fallback-img-${walk.id}').style.display='flex';">`;

            }

            

            const fallbackColor = (walk.vk_id && walk.vk_id !== "") ? '#0077FF' : (walk.youtube_id ? '#ff0000' : '#888');

            const platformName = (walk.vk_id && walk.vk_id !== "") ? 'VK Видео' : (walk.youtube_id ? 'YouTube' : 'Видео');

            

            imageHtml += `

                <div id="fallback-img-${walk.id}" class="video-fallback" style="${walk.youtube_id ? 'display:none;' : 'display:flex;'} background-color: ${fallbackColor};">

        function hideAllLines() {

            allPolylines.forEach(p => p.setStyle({ opacity: 0 }));

        }


        function renderMarkers(data) {

            if (markerClusterGroup) map.removeLayer(markerClusterGroup);

            allPolylines.forEach(poly => map.removeLayer(poly));

            

            allMarkers = [];

            allPolylines = [];


            data.forEach(walk => {

                const polyline = L.polyline(walk.route, {

                    color: '#FF5252',

                    weight: 4,             

                    opacity: 0,            

                    dashArray: '10, 10',   

                    lineCap: 'round'

                }).addTo(map);


                allPolylines.push(polyline);


                const marker = L.marker(walk.route[0]);

                

                // Создаём попап с направлением из данных

                const popup = L.popup({ 

                    autoClose: false, 

                    closeOnClick: false,

                    direction: walk.popup_direction || 'auto' // ← Используем направление из данных

                }).setContent(createPopupContent(walk));

                

                marker.bindPopup(popup);


                marker.on('mouseover', function() {

                    hideAllLines();

                    polyline.setStyle({ opacity: 1.0 });

                    polyline.bringToFront();

                    this.openPopup();

                    activePopup = popup;

                });


                marker.on('mouseout', function() {

                    setTimeout(() => {

                        const isMouseOverPopup = document.querySelector('.leaflet-popup-pane:hover');

                        if (!isMouseOverPopup) {

                            hideAllLines();

                            this.closePopup();

                            activePopup = null;

                        }

                    }, 100);

                });


                marker.on('popupopen', function() {

                    const popupPane = document.querySelector('.leaflet-popup-content-wrapper');

                    if (popupPane) {

                        popupPane.addEventListener('mouseleave', function() {

                            hideAllLines();

                            marker.closePopup();

                            activePopup = null;

                        });

                    }

                });


                allMarkers.push(marker);

            });


            markerClusterGroup = L.markerClusterGroup();

            allMarkers.forEach(m => markerClusterGroup.addLayer(m));

            map.addLayer(markerClusterGroup);


            document.getElementById('countInfo').textContent = `Показано: ${data.length}`;

        }


        renderMarkers(walks);


        map.on('click', function() {

            hideAllLines();

            map.closePopup();

            activePopup = null;

        });


        document.getElementById('streetFilter').addEventListener('input', function(e) {

            const term = e.target.value.toLowerCase();

            const filtered = walks.filter(w => w.street.toLowerCase().includes(term));

            renderMarkers(filtered);

        });
